#include <iostream>

using namespace std;

int main()
{
    int z,l,a;
    cout<<"Ziua:";cin>>z;
    cout<<"Luna:";cin>>l;
    cout<<"Anul:";cin>>a;
    cout<<"Ziua precedenta: ";
    if(z>1)z=z-1;
    else
    {
        l=l-1;
        if(l==0)
        {
            a=a-1;
            l=12;
            z=31;
        }
        else if((l==1)||(l==3)||(l==5)||(l==7)||(l==8)||(l==10))
            z=31;
        else if((l==4)||(l==6)||(l==9)||(l==11))
            z=30;
        else if (a%4==0 && a%100!=0 || a%400==0)
            z=29;
        else z=28;
    }
    cout<<z<<" "<<l<<" "<<a;
}
