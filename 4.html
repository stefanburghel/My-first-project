#include <iostream>
using namespace std;

int main()
{
   int n, nr, s=0, x, i, d, smax=0;
   cout<<"n=";cin>>n;
   for(i=1;i<=n;i++)
   {
       cout<<"x=";cin>>x;
       s=0;
       for(d=1;d<=x/2;d++)
       if(x%d==0)s=s+d;
       if(s>smax)
       {nr=x;
        smax=s;}
   }
   cout<<nr<<" "<<smax;
}
